99 static std::string sformat(const std::string& format, ...)
100 {
101  uint size = 100;
102  std::string result;
103  va_list ap;
104 
105  while (true)
106  {
107  result.resize(size);
108 
109  va_start(ap, format);
110  int n = vsnprintf((char *) result.c_str(), size, format.c_str(), ap);
111  va_end(ap);
112 
113  if (n > -1 && n < (int) size)
114  {
115  result.resize((uint) n);
116  break;
117  }
118 
119  if (n > -1)
120  {
121  size = (uint) n + 1;
122  }
123  else
124  {
125  size *= 2;
126  }
127  }
128 
129  return result;
130 }

161 std::string CLI::runCommand(std::string command)
162 {
163  // Creating string stream.
164  std::stringstream ss(command);
165 
166  // Reading command.
167  std::string cmd;
168  ss >> cmd;
169 
170  // Running processors.
171  try
172  {
173  if (cmd == "init") return this -> init(ss, false);
174  if (cmd == "random-init") return this -> init(ss, true);
175  if (cmd == "generate") return this -> generate(ss);
176  if (cmd == "create") return this -> create(ss);
177  if (cmd == "list") return this -> list(ss);
178  if (cmd == "info") return this -> info(ss);
179  if (cmd == "change") return this -> change(ss);
180  if (cmd == "step") return this -> step(ss);
181  }
182  catch(ECLIInvalidInput& exc)
183  {
184  return exc.what();
185  }
186 
187  return sformat("Unknown command `%s`.\n", cmd.c_str());
188 }

388 std::string CLI::list(std::stringstream& ss, uint columns)
389 {
390  // Reading type.
391  std::string type = "";
392  ss >> type;
393 
394  // Output.
395  std::string output;
396 
397  for (uint k = 0; k < 2; k++)
398  {
399  ObjectHeap* objs = (k == 0 ? this -> world -> getVisibleObjects() :
400  this -> world -> getHiddenObjects());
401 
402  // Preparing begin and end iterators.
403  ObjectHeap::const_iterator iter_begin, iter_end;
404  if (type == "")
405  {
406  iter_begin = objs -> begin();
407  iter_end = objs -> end();
408  }
409  else if (type == TN_BUILDING)
410  {
411  iter_begin = objs -> begin(BUILDING);
412  iter_end = objs -> end(BUILDING);
413  }
414  else if (type == TN_CREATURE)
415  {
416  iter_begin = objs -> begin(CREATURE);
417  iter_end = objs -> end(CREATURE);
418  }
419  else if (type == TN_RESOURCE)
420  {
421  iter_begin = objs -> begin(RESOURCE);
422  iter_end = objs -> end(RESOURCE);
423  }
424  else if (type == TN_WEATHER)
425  {
426  iter_begin = objs -> begin(WEATHER);
427  iter_end = objs -> end(WEATHER);
428  }
429  else
430  {
431  return sformat("Error: unknown ObjectType (%s).\n", type.c_str());
432  }
433 
434  // Listing objects.
435  ObjectHeap::const_iterator iter;
436  uint cur_column = 1;
437  for (iter = iter_begin; iter != iter_end; iter++)
438  {
439  Object* obj = *iter;
440 
441  // Preparing flags.
442  std::string flags = k == 0 ? "v" : "h";
443  flags += obj -> isDestroyed() ? "d" : " ";
444  flags += obj -> isImmortal() ? "i" : " ";
445  flags += obj -> isSolid() ? "s" : " ";
446 
447  // Printing output.
448  output += sformat("%d\t %s %s %f\t%f", obj -> getObjectID(),
449  flags.c_str(),
450  obj -> getTypeName().c_str(),
451  obj -> getCoords().getX(),
452  obj -> getCoords().getY());
453 
454  if (cur_column++ == columns)
455  {
456  output += "\n";
457  cur_column = 1;
458  }
459  else
460  {
461  output += "\t\t\t";
462  }
463  }
464  }
465 
466  return output + "\n";
467 }
